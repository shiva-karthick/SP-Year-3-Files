 Dim cstype As Type
        cstype = Me.GetType()

        ' Create a message string 
        Dim msg1 As String
        'msg1 = "<script>window.status='This is a message in the status bar';</script>"

        msg1 = "<script>window.status='The Date and Time is " & DateTime.Now.ToString("F") & ".';</script>"

        ' Register startup script to show a message in status bar
        ClientScript.RegisterStartupScript(cstype, "SetStatusText", msg1)
